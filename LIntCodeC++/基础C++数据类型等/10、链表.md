## 1、定义
链表：由指针连接的结构体。
```C++
struct Node{
	int data;
	Node* next;
};
```
补充：存储期[[存储期（分配内存~释放内存）]]
## 2、操作
### 1、创建
```c++
Node* creatlist(int n)
{
	// Node* head{new Node{0}};
	Node* head  = new Node;
	head->next = NULL;
	Node* pre = head;
	for(int i = 0;i<n;i++)
	{
		Node* current = new Node;
		current->data = i+1;
		pre->next = current;
		pre = current;
		current->next = NULL;
	}
	return head;
}
```
### 2、遍历
```C++
void ShowLinkList(Node* head)
{
	Node* current = head->next;
	int e;
	while(current!=NULL)
	{
		e = current->data;
		cout<<e<<" ";
		current = current->next;
	}
}
```
### 3、插入/删除元素
```C++
Node* find(Node* head,int n)
{
	for(int i = 0;i<n;i++)
	{
		head = head->next;
	}
	return head;
}
//指定位置n插入元素e
Node* Insert(Node* head,int e,int n)
{
	Node* current = new Node;
	current->data = e;
	Node* pre = find(head,n-1);
	current->next = pre->next;
	pre->next = current;
	return head;
}
//指定位置删除元素e
Node* Delete(Node* head,int n)
{
	Node* pre = find(head,n-1);
	Node* del = pre->next;
	pre->next  = del->next;
	delete del;
}
```
